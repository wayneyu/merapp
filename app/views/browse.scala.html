@for( y <- allyears){ } @for( c <- allcourses ){ @for( y <- allyears ){ @{ val dec = {if( exams.contains( (c, y, "December") )) "Dec" else "---"} val apr = {if( exams.contains( (c, y, "April") )) "Apr" else "---"} } } }
@y

}
